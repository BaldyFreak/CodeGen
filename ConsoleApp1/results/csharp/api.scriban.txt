import request from '@/utils/request'

// 查询参数列表
export function listTest(query) {
  return request({
    url: '/system/test/list',
    method: 'get',
    params: query
  })
}

// 查询参数详细
export function getTest(id) {
  return request({
    url: '/system/test/' + id,
    method: 'get'
  })
}

// 新增参数配置
export function addTest(data) {
  return request({
    url: '/system/test',
    method: 'post',
    data: data
  })
}

// 修改参数配置
export function updateTest(data) {
  return request({
    url: '/system/test',
    method: 'put',
    data: data
  })
}

// 删除参数配置
export function delTest(id) {
  return request({
    url: '/system/test/' + id,
    method: 'delete'
  })
}
