using Ardalis.Specification;
using Garen.MultiTenant.Core.Entities;
using Microsoft.EntityFrameworkCore;
using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using static Microsoft.EntityFrameworkCore.DbLoggerCategory;

namespace Garen.MultiTenant.Service.Admin.Specifications
{
    public class SysTestSpecification : Specification<SysTest>
    {
        public SysTestSpecification(SysTest test)
        {
            
             if (test.TestId != 0)
            {
                Query.Where(x => x.TestId == test.TestId);
            }
            if (!string.IsNullOrEmpty(test.TestName))
            {
                Query.Where(x => EF.Functions.Like(x.TestName, $"%{ test.TestName }%"));
            }
            if (!string.IsNullOrEmpty(test.CreateTime))
            {
                Query.Where(x => x.CreateTime == test.CreateTime);
            }
        }
    }
}
