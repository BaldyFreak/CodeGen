using Garen.MultiTenant.Service.Admin;
using Garen.MultiTenant.Api.ModelBinders;
using Garen.MultiTenant.Core.Dtos;
using Garen.MultiTenant.Core.Entities;
using Garen.MultiTenant.Infrastructure.Database.DbContexts;
using Microsoft.AspNetCore.Mvc;
using Microsoft.EntityFrameworkCore;
using MiniExcelLibs;
using System.Web;

namespace Garen.MultiTenant.Api.Controllers
{
    [Route("admin/test")]
    [ApiController]
    public class SysTestController : ControllerBase
    {
        private readonly SysTestService _sysTestService;
        private readonly IWebHostEnvironment _env;
        public SysTestController(SysTestService sysTestService, IWebHostEnvironment environment)
        {
            _sysTestService = sysTestService;
            _env = environment;
        }
        [HttpGet("list")]
        public async Task<IActionResult> GetPage([FromQuery] SysTest test)
        {
            var list = await _sysTestService.GetPage(test);
            var result = new TableDataInfo<SysTest>(list, list.Count);
            return Ok(result);

        }
        [HttpGet("{id}")]
        public async Task<IActionResult> GetById(long id)
        {
            var test = await _sysTestService.GetByIdAsync(id);
            return Ok(new { code = 200, msg = "操作成功", data = test });
        }

        [HttpPost]
        public async Task<IActionResult> Add([FromBody] SysTest test)
        {
            await _sysTestService.AddAsync(test);
            return Ok(new { code = 200, msg = "成功" });
        }

        [HttpPut]
        public async Task<IActionResult> Update([FromBody] SysTest test)
        {
            await _sysTestService.UpdateAsync(test);
            return Ok(new { code = 200, msg = "成功" });
        }
        [HttpDelete("{ids}")]
        public async Task<IActionResult> Delete([ModelBinder(typeof(QueryArray<long>))] long[] ids)
        {
            await _sysTestService.DeleteRange(ids);
            return Ok(new { code = 200, msg = "成功" });
        }
    }
}
